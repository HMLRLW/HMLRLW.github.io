What is huffman tree?

赫夫曼树就是从一组节点中取出最小得那两个，然后将小得放左边，大的放右边。二者相加的值得到新的节点为他们的父节点。将该节点放到我们节点集合里面，然后依次递归上面的过程。

package interview.day1025;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

/**
 * 赫夫曼树的使用
 *
 * @author liwei
 * @date 2019/10/25 20:08
 */
public class HefumanTreeDemo {

    public static void main(String[] args) {
        int[] arrays = {1, 2, 4, 3, 5, 7, 6, 9, 3, 10, 79, 86, 12, 43, 56, 24, 78};
        HefumanTree.hefumanTree(arrays);
        System.out.println(1 + 2 + 4 + 3 + 5 + 7 + 6 + 9 + 3 + 10 + 79 + 86 + 12 + 43 + 56 + 24 + 78);
    }
}

class HefumanTree {


    /**
     * 封装对外的方法
     *
     * @param arrays
     * @return
     */
    public static int[] hefumanTree(int[] arrays) {
        List<Node> nodes = createNode(arrays);
        createHefumanTree(nodes);
        Node node = nodes.get(0);
        System.out.println(node.toString());
        return null;
    }

    /**
     * 创建节点.
     *
     * @param arrays
     * @return
     */
    public static List<Node> createNode(int[] arrays) {
        List<Node> nodes = new ArrayList<>(arrays.length);
        for (int i = 0; i < arrays.length; i++) {
            Node node = new Node();
            node.value = arrays[i];
            nodes.add(node);
        }
        return nodes;
    }

    /**
     * 创建赫夫曼树.
     *
     * @param nodes
     */
    public static void createHefumanTree(List<Node> nodes) {
        if (nodes == null || nodes.size() == 1) {
            return;
        }
        // 排序
        Collections.sort(nodes);

        Node node1 = nodes.remove(0);
        Node node2 = nodes.remove(0);

        Node node = new Node();
        node.value = node1.value + node2.value;
        if (node1.value < node2.value) {
            node.left = node1;
            node.right = node2;
        } else {
            node.left = node2;
            node.right = node1;
        }
        nodes.add(node);
        createHefumanTree(nodes);
    }
}


class Node implements Comparable<Node> {
    public int value;
    public Node left;
    public Node right;

    @Override
    public int compareTo(Node o) {
        return this.value > o.value ? -1 : 1;
    }

    @Override
    public String toString() {
        return "Node{" +
                "value=" + value +
                ", left=" + left +
                ", right=" + right +
                '}';
    }
}
